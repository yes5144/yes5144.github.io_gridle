tmux

## 安装
yum  install tmux  -y

## 使用
tmux
[root@justdoit ~]# tmux
[root@justdoit ~]# tmux ls
0: 1 windows (created Sun Jun 30 00:32:05 2019) [164x26] (attached)
[root@justdoit ~]# tmux rename -t 0 test_tmux
[root@justdoit ~]# tmux ls
test_tmux: 1 windows (created Sun Jun 30 00:32:05 2019) [164x26] (attached)
[root@justdoit ~]#
