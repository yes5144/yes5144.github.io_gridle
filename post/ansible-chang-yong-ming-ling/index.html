- hosts: mfs_node
  tasks:
    - shell: echo "5"
      register: result
      ignore_errors: True

    - debug: msg="it failed"
      when: result|failed

    - debug: msg="{{result.stdout}}"

    - shell: /usr/bin/var
      when: result.rc == 5

# A YAML Gotcha
# this won't work:

- host: app_servers
  vars:
    app_path: {{ base_path }}/22

# Do it like this and you'll be fine:

- hosts: app_servers
  vars:
    app_path: "{{ base_path }}/22"  # 加上双引号
